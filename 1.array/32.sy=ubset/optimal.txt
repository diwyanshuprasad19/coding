
string isSubset(int a1[], int a2[], int n, int m) {
    unordered_set<int>s;
    for(int i=0;i<n;i++)
    s.insert(a1[i]);
    int count=0;
    for(int j=0;j<m;j++)
    {
        if(s.find(a2[j])!=s.end())
        {
            count++;
            
        }
    }
    if(count==m)
    return "Yes";
    else
    return "No";
    
}
