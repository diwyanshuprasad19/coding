    vector<int> bfsOfGraph(int V, vector<int> adj[]) {
           vector<int > v;
    queue<int>q ;
    vector<int>vis(V,0);
    vis[0]=1;
    q.push(0);
    while(q.empty()==false)
    {
        int temp=q.front();
        q.pop();
        v.push_back(temp);
        for(int i=0;i<adj[temp].size();i++)
        {
            if(vis[adj[temp][i]]==0)
            {
                vis[adj[temp][i]]=1;
                q.push(adj[temp][i]);
            }
        }
    }
    return v;
    }
