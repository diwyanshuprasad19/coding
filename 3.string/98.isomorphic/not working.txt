#include <bits/stdc++.h>


using namespace std;

int main()
{
   string s="aabbbccccd";
   string p="xyyzzzqqqq";
   unordered_map<char,int>m1;
   unordered_map<char,int>m2;
   int n=s.length();
   int m=p.length();
   for(int i=0;i<n;i++)
   m1[s[i]]++;
    for(int i=0;i<m;i++)
   m2[p[i]]++;
   int flag=1;
   if(m1.size()==m2.size())
   {
       for(auto itr1=m1.begin();itr1!=m1.end();itr1++)
       {
           for(auto itr2=m2.begin();itr2!=m2.end();itr2++)
           {
               if(itr1->second==itr2->second)
               {
                   itr2->second=0;
                   break;
               }
           }
       }
       for(auto itr1=m2.begin();itr1!=m2.end();itr1++)
       {
           if(itr1->second!=0)
           flag=0;
       }
   }
   else
   {
      flag=0;
   }
   
   if(flag==1)
   cout<<"true";
   else
   cout<<"false";
   
   
   
   return 0;
}
