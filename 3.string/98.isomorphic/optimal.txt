    bool areIsomorphic(string str1, string str2)
    {
        
        int n=str1.length();
        int m=str2.length();
        if(n!=m) return false;
        int n1[256]={0};
        int m1[256]={0};
        for(int i=0;i<n;i++)
        {
            if(!n1[str1[i]]  && !m1[str2[i]])
            {
                n1[str1[i]]=str2[i];
                m1[str2[i]]=str1[i];
            }
            else if(n1[str1[i]]!=str2[i])return false;
        }
        return true;
        
    }
