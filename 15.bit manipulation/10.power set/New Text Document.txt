class Solution{
	public:
		vector<string> AllPossibleStrings(string s){
		    int n=s.size();
		    int n1=pow(2,n);
		    vector<string> ans;
		    string s1;
		    for(int i=0;i<n1;i++)
		    {
		        for(int j=0;j<n;j++)
		        {
		            if(i &(1<<j))
		            s1=s1+s[j];
		        }
		        ans.push_back(s1);
		        s1="";
		    }
		    sort(ans.begin(),ans.end());
		    return ans;
		}
};
