#include<bits/stdc++.h>
int maxMoneyLooted(vector<int> &houses, int n)
{
	vector<int> dp(n,-1);
    dp[0]=houses[0];
    int neg=0;
    for(int i=1;i<n;i++)
    {
        int pick=houses[i];
        if(i>1)pick+=dp[i-2];
        int notpick=0+dp[i-1];
        dp[i]=max(pick,notpick);
    }
    return dp[n-1];
}
