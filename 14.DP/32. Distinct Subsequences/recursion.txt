#include<bits/stdc++.h>

int solve(int i,int j,string s1,string s2)
{
    if(j<0)return 1;
    if(i<0)return 0;
    if(s1[i]==s2[j])
    {
        return (solve(i-1,j-1,s1,s2)+solve(i-1,j,s1,s2));
    }
    else
        return solve(i-1,j,s1,s2);
}

int subsequenceCounting(string &t, string &s, int lt, int ls) {
    int count=solve(lt-1,ls-1,t,s);
    return count;
}
