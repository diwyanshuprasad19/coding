#include<bits/stdc++.h>
int solve(int index,int last, vector<vector<int>> &points)
{
    if(index==0)
    {
        int maxi=INT_MIN;
        for(int i=0;i<3;i++)
        {
            if(i!=last)
                maxi=max(maxi,points[0][i]);
        }
        return maxi;
    }
    int maxi=INT_MIN;
    for(int i=0;i<3;i++)
    {
        if(i!=last)
        {
            int p=points[index][i]+solve(index-1,i,points);
            maxi=max(maxi,p);
        }
    }
    return maxi;
}

int ninjaTraining(int n, vector<vector<int>> &points)
{
   int m=solve(n-1,3,points);
    return m;
}
