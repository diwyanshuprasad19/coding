#include <bits/stdc++.h> 
int maximumProfit(vector<int> &prices){
    int mini=prices[0];
    int n=prices.size();
    int maxprofit=0;
    for(int i=1;i<n;i++)
    {
        mini=min(prices[i],mini);
        int cost=prices[i]-mini;
        maxprofit=max(maxprofit,cost);
    }
    return maxprofit;
}

time:O(n)
space:1
