#include<bits/stdc++.h>
int solve(int i,int j,vector<int> &arr)
{
    if(i==j)return 0;
    int mini=1e9;

    for(int k=i;k<=j-1;k++)
    {
         int steps=(arr[i-1]*arr[k]*arr[j])+solve(i,k,arr)+solve(k+1,j,arr);
        if(steps<mini) mini=steps;
    }
    return mini;
}
int matrixMultiplication(vector<int> &arr, int N)
{
    int count=solve(1,N-1,arr);
    return count;
}
