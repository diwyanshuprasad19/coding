#include<bits/stdc++.h>
int solve(int i, int j, vector< vector< int> > &mat)
{
    if(i>=0 && j>=0 && mat[i][j]==-1)return 0;
    if(i==0 && j==0)return 1;
if(i<0 || j<0)return 0;
    int up=solve(i-1,j,mat);
    int left=solve(i,j-1,mat);
    return up+left;
}
int mazeObstacles(int n, int m, vector< vector< int> > &mat) {
    int count=solve(n-1,m-1,mat);
    return count;
}
