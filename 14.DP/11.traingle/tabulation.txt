#include<bits/stdc++.h>
int minimumPathSum(vector<vector<int>>& triangle, int n){
vector<vector<int>> dp(n,vector<int>(n,-1));
    for(int i=0;i<n;i++)
    {
        dp[n-1][i]=triangle[n-1][i];
    }
    for(int i=n-2;i>=0;i--)
    {
        for(int j=i;j>=0;j--)
        {
            int down=triangle[i][j]+dp[i+1][j];
            int diagnol=triangle[i][j]+dp[i+1][j+1];
            dp[i][j]=min(down,diagnol);
        }
    }
    return dp[0][0];
}
