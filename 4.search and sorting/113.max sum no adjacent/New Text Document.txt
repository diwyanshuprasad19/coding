    int dp[1000000];
    int solve(int arr[],int n)
    {
        if(n<=-1)return 0;
        if(dp[n]!=-1)return dp[n];
        int n1=arr[n]+solve(arr,n-2);
        int n2=solve(arr,n-1);
        return dp[n]=max(n1,n2);
    }
    int FindMaxSum(int arr[], int n)
    {

       memset(dp,-1,1000000);
       int m=solve(arr,n-1);
       return m;
    }
