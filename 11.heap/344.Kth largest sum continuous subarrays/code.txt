    int kthLargest(vector<int> &Arr,int N,int K){
        priority_queue<int,vector<int>,greater<int>> p;
        for(int i=0;i<N;i++)
        {
            int ans=0;
            for(int j=i;j<N;j++)
            {
                ans+=Arr[j];
                p.push(ans);
                if(p.size()>K)
                p.pop();
            }
        }
        return p.top();
    }
