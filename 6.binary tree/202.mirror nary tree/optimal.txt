    int checkMirrorTree(int n, int e, int A[], int B[]) {
        unordered_map<int,stack<int>>m;
        for(int i=0;i<2*e;i=i+2)
        {
            m[A[i]].push(A[i+1]);
        }
        for(int j=0;j<2*e;j=j+2)
        {
            if(m[B[j]].top()!=B[j+1])return 0;
            m[B[j]].pop();
        }
        return 1;
