Node* reverseDLL(Node * head)
{
        Node* curr =head;
    if(curr==NULL || curr->next==NULL)
    return head;
    while(curr->next!=NULL)
    curr=curr->next;
    head=curr;
    curr->next=curr->prev;     //first manually
    curr->prev=NULL;
    curr=curr->next;
    while(curr->prev!=NULL)
    {
        Node* temp=curr->prev;
        curr->prev=curr->next;
        curr->next=temp;
        curr=curr->next;
    }
    //last manually too
    curr->prev=curr->next;
    curr->next=NULL;
    
    return head;
}
