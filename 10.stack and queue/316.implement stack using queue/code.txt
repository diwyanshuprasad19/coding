void QueueStack :: push(int x)
{
q1.push(x);
while(q2.empty()!=true)
{
    q1.push(q2.front());
    q2.pop();
}
swap(q1,q2);
}

//Function to pop an element from stack using two queues. 
int QueueStack :: pop()
{
    int ans;
        if(q2.empty())
        {
            ans=-1;
        }
        else
        {
            ans=q2.front();
            q2.pop();
        }
        return ans;
}
